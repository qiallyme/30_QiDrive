> ## 4. Python script to build the .ico files & zip the pack
> 
> Once you’ve generated your **PNG art** (using the prompts from earlier), drop them into `source_png/`, then run this script:
> 
> ```python
> import os
> import zipfile
> from datetime import datetime
> 
> from PIL import Image  # pip install pillow
> 
> 
> ICON_SIZES = [16, 24, 32, 48, 64, 128, 256]
> 
> ICON_MAP = {
>     "qios_root": "qios_root.png",
>     "qivault": "qivault.png",
>     "qinote": "qinote.png",
>     "qidocs": "qidocs.png",
>     "qiprojects": "qiprojects.png",
>     "qiclients": "qiclients.png",
>     "qiarchive": "qiarchive.png",
>     "qios_engine": "qios_engine.png",
>     "qiapps": "qiapps.png",
>     "meta": "meta.png",
>     "system": "system.png",
>     "private": "private.png",
>     "inbox": "inbox.png",
>     "hotcache": "hotcache.png",
>     "templates": "templates.png",
>     "assets": "assets.png",
>     "repos": "repos.png",
> }
> 
> ROOT_DIR = os.path.dirname(os.path.abspath(__file__))
> SRC_DIR = os.path.join(ROOT_DIR, "source_png")
> ICO_DIR = os.path.join(ROOT_DIR, "ico")
> OUT_ZIP = os.path.join(ROOT_DIR, f"QiOS_IconPack_{datetime.now().strftime('%Y%m%d')}.zip")
> 
> 
> def ensure_dirs():
>     os.makedirs(ICO_DIR, exist_ok=True)
> 
> 
> def build_ico(png_name: str, ico_name: str):
>     src_path = os.path.join(SRC_DIR, png_name)
>     dst_path = os.path.join(ICO_DIR, ico_name)
> 
>     if not os.path.exists(src_path):
>         print(f"[WARN] Missing source PNG: {src_path}")
>         return
> 
>     img = Image.open(src_path).convert("RGBA")
>     sizes = [(s, s) for s in ICON_SIZES]
>     img.save(dst_path, format="ICO", sizes=sizes)
>     print(f"[OK] {dst_path}")
> 
> 
> def build_all_icons():
>     print("=== Building ICO files ===")
>     for key, png in ICON_MAP.items():
>         ico_name = f"{key}.ico"
>         build_ico(png, ico_name)
> 
> 
> def build_zip():
>     print(f"\n=== Building ZIP: {OUT_ZIP} ===")
>     with zipfile.ZipFile(OUT_ZIP, "w", zipfile.ZIP_DEFLATED) as zf:
>         for folder_name, subdirs, filenames in os.walk(ROOT_DIR):
>             rel_folder = os.path.relpath(folder_name, ROOT_DIR)
>             if rel_folder == ".":
>                 rel_folder = ""
>             for filename in filenames:
>                 if filename == os.path.basename(__file__):
>                     continue
>                 file_path = os.path.join(folder_name, filename)
>                 arcname = os.path.join(rel_folder, filename)
>                 zf.write(file_path, arcname)
>     print("[DONE] ZIP created.")
> 
> 
> if __name__ == "__main__":
>     ensure_dirs()
>     build_all_icons()
>     build_zip()
> ```
> 
> **What this does:**
> 
> 1. Looks in `source_png/` for the expected PNG filenames.  
> 2. Creates multi-size `.ico` files with all classic icon sizes.  
> 3. Packs the **entire QiOS_IconPack folder** into a versioned zip like:  
>    `QiOS_IconPack_20251126.zip`.
> 
> Run it with:
> 
> ```bash
> python build_icons.py
> ```

> import os

> 
> # ==== CONFIG ====
> INPUT_DIR = r"C:\Users\codyr\OneDrive\Pictures\Screenshots\iconset"
> OUTPUT_DIR = r"C:\Users\codyr\OneDrive\Pictures\Screenshots\iconset_clean"

> MAKE_SQUARE = True
> FINAL_SIZE = None  # e.g. 512 if you want to force everything to 512x512
> # =================
> 
> 
> def color_distance(c1, c2):
>     return sum((a - b) ** 2 for a, b in zip(c1[:3], c2[:3])) ** 0.5
> 
> 
> def guess_background_color(img):
>     """
>     Use the 4 corners to guess the background color.
>     """
>     pixels = img.load()
>     w, h = img.size
>     corners = [
>         pixels[0, 0],
>         pixels[w - 1, 0],
>         pixels[0, h - 1],
>         pixels[w - 1, h - 1],
>     ]
>     # Simple average of corners
>     r = sum(p[0] for p in corners) // 4
>     g = sum(p[1] for p in corners) // 4
>     b = sum(p[2] for p in corners) // 4
>     return (r, g, b, 255)
> 
> 
> def remove_background(img, tolerance=20):
>     """
>     Turn background pixels (similar to guessed bg color) into transparent.
>     """
>     img = img.convert("RGBA")
>     bg = guess_background_color(img)
> 
>     pixels = img.load()
>     w, h = img.size
> 
>     for y in range(h):
>         for x in range(w):
>             r, g, b, a = pixels[x, y]
>             if a == 0:
>                 continue
>             d = color_distance((r, g, b, a), bg)
>             if d <= tolerance:
>                 pixels[x, y] = (r, g, b, 0)
> 
>     return img
> 
> 
> def make_square_canvas(img):
>     """
>     Put the image on a square transparent canvas, centered.
>     """
>     w, h = img.size
>     size = max(w, h)
>     square = Image.new("RGBA", (size, size), (0, 0, 0, 0))
>     offset = ((size - w) // 2, (size - h) // 2)
>     square.paste(img, offset, img)
>     return square
> 
> 
> def maybe_resize(img, target_size):
>     if target_size is None:
>         return img
>     return img.resize((target_size, target_size), Image.LANCZOS)
> 
> 
> def process_icon(path: Path, out_dir: Path):
>     print(f"[+] Processing {path.name}")
>     img = Image.open(path)
> 
>     # Step 1: remove background
>     img = remove_background(img, tolerance=BACKGROUND_TOLERANCE)
> 
>     # Step 2: square it if needed
>     if MAKE_SQUARE:
>         img = make_square_canvas(img)
> 
>     # Step 3: optional resize
>     img = maybe_resize(img, FINAL_SIZE)
> 
>     # Step 4: save
>     out_dir.mkdir(parents=True, exist_ok=True)
>     out_path = out_dir / path.name
>     img.save(out_path, format="PNG")
>     print(f"    -> saved to {out_path}")
> 
>

>     out_dir = Path(OUTPUT_DIR)
>

> 
>     pngs = list(in_dir.glob("*.png"))
>     if not pngs:
>         print(f"No PNG files found in {in_dir}")
>         return
> 
>     for p in pngs:
>         process_icon(p, out_dir)
> 
>     print("\nDone. Go admire your cleaned icons.")
> 
>

>     main()

> def slugify(stem: str) -> str:
>     """
>     Normalize a filename stem:
>     - lowercase
>     - replace spaces, parentheses, hyphens with underscores
>     - remove anything not alphanumeric or underscore
>     - collapse multiple underscores
>     """
>     s = stem.lower()
>     s = s.replace(" ", "_").replace("-", "_")
>     s = s.replace("(", "_").replace(")", "_")
>     # remove other junk but keep underscores
>     s = re.sub(r"[^a-z0-9_]+", "_", s)
>     # collapse multiple underscores
>     s = re.sub(r"_+", "_", s)
>     # strip leading/trailing underscores
>     s = s.strip("_")
>     return s or "icon"
> 
> 
> def rename_pair(base_dir: Path, ico_dir: Path, img_path: Path, dry_run: bool = False):
>     stem = img_path.stem
>     ext = img_path.suffix.lower()
>     new_stem = slugify(stem)
> 
>     if new_stem == stem:
>         # no change
>         return
> 
>     new_img_path = img_path.with_name(new_stem + ext)
> 
>     # Check ICO counterpart
>     ico_path = ico_dir / (stem + ".ico")
>     new_ico_path = ico_dir / (new_stem + ".ico")
> 
>     # Collision check
>     if new_img_path.exists() or (ico_path.exists() and new_ico_path.exists()):
>         print(f"[SKIP] Would collide: {img_path.name} -> {new_img_path.name}")
>         return
> 
>     print(f"[RENAME] {img_path.name} -> {new_img_path.name}")
>     if ico_path.exists():
>         print(f"         {ico_path.name} -> {new_ico_path.name}")
> 
>     if not dry_run:
>         img_path.rename(new_img_path)
>         if ico_path.exists():
>             ico_path.rename(new_ico_path)
> 
> 
> def main():
>     if not ROOT_DIR.exists():
>         print(f"Root dir not found: {ROOT_DIR}")
>         return
> 
>     ico_dir = ROOT_DIR / ICO_DIR_NAME

>         print(f"ICO subfolder not found at {ico_dir}")
>         return

>         return
