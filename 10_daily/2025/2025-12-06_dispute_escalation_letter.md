> ## 2️⃣ `auto_sort_aes.py` – ready-to-use sorter
> 
> ```python
> #!/usr/bin/env python3
> import argparse
> import datetime as dt
> import os
> import re
> import shutil
> from pathlib import Path
> 
> CATEGORY_MAP = {
>     "Bill": "30_Evidence_and_Exhibits/Bills_and_Statements",
>     "Meter": "30_Evidence_and_Exhibits/Meter_Readings_and_Photos",
>     "Notice": "30_Evidence_and_Exhibits/Disconnection_Notices",
>     "ADA": "30_Evidence_and_Exhibits/ADA_Documents",
>     "Correspondence": "20_Correspondence/AES_Correspondence",
>     "Evidence": "30_Evidence_and_Exhibits/Public_Evidence",
>     "Affidavit": "30_Evidence_and_Exhibits/Affidavits_and_Witness_Statements",
>     "Log": "30_Evidence_and_Exhibits/Logs",
>     "Filing": "10_Filings_and_Regulatory_Actions",
>     "Packet": "70_Packets_and_Assemblies",
>     "Strategy": "40_Analysis_and_Findings",
>     "Export": "80_Final_Deliverables",
>     "Audio": "90_Visuals_and_Audio/Recorded_Calls_and_Notes",
>     "Canvas": "90_Visuals_and_Audio/Obsidian_Canvas_Maps",
> }
> 
> FOLDER_PREFIXES = tuple(f"{i:02d}_" for i in range(0, 100))
> SKIP_NAMES = {
>     "index.md",
>     "naming_rules.md",
>     "auto_sort_aes.py",
> }
> SKIP_DIRS = {
>     ".git",
>     ".obsidian",
>     "__pycache__",
> }
> 
> FILENAME_RE = re.compile(
>     r"^(?P<date>\d{4}-\d{2}-\d{2})_(?P<category>[A-Za-z]+)_(?P<rest>.+)$"
> )
> 
> 
> def log_action(root: Path, message: str) -> None:
>     log_dir = root / "00_Admin_and_Overview"
>     log_dir.mkdir(parents=True, exist_ok=True)
>     log_file = log_dir / "auto_sort_log.txt"
>     timestamp = dt.datetime.now().strftime("%Y-%m-%d %H:%M:%S")
>     with log_file.open("a", encoding="utf-8") as f:
>         f.write(f"[{timestamp}] {message}\n")
> 
> 
> def is_already_sorted(root: Path, path: Path) -> bool:
>     try:
>         rel = path.relative_to(root)
>     except ValueError:
>         return False
>     parts = rel.parts
>     if not parts:
>         return False
>     return parts[0].startswith(FOLDER_PREFIXES)
> 
> 
> def classify_file(path: Path):
>     name = path.name
>     m = FILENAME_RE.match(name)
>     if not m:
>         return None
>     data = m.groupdict()
>     category = data["category"]
>     target_rel = CATEGORY_MAP.get(category)
>     if not target_rel:
>         return None
>     return target_rel
> 
> 
> def unique_destination(dest: Path) -> Path:
>     if not dest.exists():
>         return dest
>     stem = dest.stem
>     suffix = dest.suffix
>     parent = dest.parent
>     counter = 1
>     while True:
>         candidate = parent / f"{stem}({counter}){suffix}"
>         if not candidate.exists():
>             return candidate
>         counter += 1
> 
> 
> def iter_candidate_files(root: Path, inbox: Path | None):
>     # root-level files
>     for item in root.iterdir():
>         if item.is_dir():
>             continue
>         if item.name in SKIP_NAMES:
>             continue
>         if item.name.startswith("."):
>             continue
>         yield item
> 
>     # inbox files if given and exists
>     if inbox and inbox.exists() and inbox.is_dir():
>         for dirpath, dirnames, filenames in os.walk(inbox):
>             # prune skip dirs
>             dirnames[:] = [d for d in dirnames if d not in SKIP_DIRS]
>             for filename in filenames:
>                 p = Path(dirpath) / filename
>                 if p.name.startswith("."):
>                     continue
>                 yield p
> 
> 
> def main():
>     parser = argparse.ArgumentParser(
>         description="Auto-sort AES utility dispute files into structured folders."
>     )
>     parser.add_argument(
>         "--root",
>         type=str,
>         default=".",
>         help="Project root directory (default: current directory)",
>     )
>     parser.add_argument(
>         "--inbox",
>         type=str,
>         default=None,
>         help="Optional inbox directory to scan for unsorted files.",
>     )
>     parser.add_argument(
>         "--dry-run",
>         action="store_true",
>         help="Do not move files, just report what would happen.",
>     )
>     args = parser.parse_args()
> 
>     root = Path(args.root).resolve()
>     inbox = Path(args.inbox).resolve() if args.inbox else None
> 
>     if not root.exists():
>         raise SystemExit(f"Root directory does not exist: {root}")
> 
>     for file_path in iter_candidate_files(root, inbox):
>         # skip if already under a 00_/10_/… dir
>         if is_already_sorted(root, file_path):
>             continue
> 
>         rel = None
>         try:
>             rel = file_path.relative_to(root)
>         except ValueError:
>             rel = file_path
> 
>         classification = classify_file(file_path)
>         if not classification:
>             msg = f"SKIP  {rel}  (unclassified: filename does not match convention or unknown category)"
>             print(msg)
>             log_action(root, msg)
>             continue
> 
>         target_dir = root / classification
>         target_dir.mkdir(parents=True, exist_ok=True)
>         dest = unique_destination(target_dir / file_path.name)
> 
>         msg = f"MOVE  {rel}  ->  {dest.relative_to(root)}"
>         if args.dry_run:
>             print("[DRY-RUN]", msg)
>             log_action(root, f"[DRY-RUN] {msg}")
>         else:
>             dest.parent.mkdir(parents=True, exist_ok=True)
>             shutil.move(str(file_path), str(dest))
>             print(msg)
>             log_action(root, msg)
> 
> 
> if __name__ == "__main__":
>     main()
> ```
> 
> Run from project root:
> 
> ```bash
> python auto_sort_aes.py --dry-run
> python auto_sort_aes.py
> ```

> ## 3️⃣ PowerShell structure-check / repair script
> 
> Save as `check_structure.ps1` in the project root:
> 
> ```powershell
> $root = Get-Location
> 
> $folders = @(
>   "00_Admin_and_Overview",
>   "10_Filings_and_Regulatory_Actions",
>   "20_Correspondence",
>   "30_Evidence_and_Exhibits",
>   "40_Analysis_and_Findings",
>   "50_Consumer_Protections",
>   "60_AES_Direct_Actions",
>   "70_Packets_and_Assemblies",
>   "80_Final_Deliverables",
>   "90_Visuals_and_Audio",
>   "99_Archive"
> )
> 
> Write-Host "Checking AES project structure in $root`n"
> 
> foreach ($f in $folders) {
>     $path = Join-Path $root $f
>     if (-not (Test-Path $path)) {
>         Write-Host "MISSING: $f  -> creating..." -ForegroundColor Yellow
>         New-Item -ItemType Directory -Path $path | Out-Null
>     } else {
>         Write-Host "OK:       $f" -ForegroundColor Green
>     }
> 
>     $readme = Join-Path $path "README.md"
>     if (-not (Test-Path $readme)) {
>         Write-Host "  + missing README.md -> creating placeholder" -ForegroundColor Yellow
>         New-Item -ItemType File -Path $readme | Out-Null
>     }
> }
> 
> # Ensure top-level index files
> $index = Join-Path $root "index.md"
> if (-not (Test-Path $index)) {
>     Write-Host "Creating top-level index.md" -ForegroundColor Yellow
>     New-Item -ItemType File -Path $index | Out-Null
> }
> 
> $naming = Join-Path $root "naming_rules.md"
> if (-not (Test-Path $naming)) {
>     Write-Host "Creating naming_rules.md" -ForegroundColor Yellow
>     New-Item -ItemType File -Path $naming | Out-Null
> }
> 
> Write-Host "`nStructure check complete."
> ```
> 
> Run from project root in PowerShell:
> 
> ```powershell
> .\check_structure.ps1
> ```
