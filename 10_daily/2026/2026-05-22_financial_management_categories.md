>

> """
> 
> import argparse
> import csv
> import os
> import sys
> from typing import Dict, List
> 
> import requests
> 
> 
> def normalize_base_url(url: str) -> str:
>     return url.rstrip("/")
> 
> 
> def get_existing_categories(base_url: str, headers: Dict[str, str]) -> Dict[str, str]:
>     """
>     Returns existing categories as:
>     {
>       "Category Name": "category_id"
>     }
>     """
>     categories = {}
>     page = 1
> 
>     while True:
>         url = f"{base_url}/api/v1/categories"
>         response = requests.get(url, headers=headers, params={"page": page}, timeout=30)
> 
>         if response.status_code == 302:
>             raise RuntimeError(
>                 "Got HTTP 302 redirect. Firefly is sending the API request to login. "
>                 "Check FIREFLY_URL, reverse proxy config, and that your token is a Personal Access Token."
>             )
> 
>         response.raise_for_status()
>         payload = response.json()
> 
>         for item in payload.get("data", []):
>             attributes = item.get("attributes", {})
>             name = attributes.get("name")
>             if name:
>                 categories[name] = item.get("id")
> 
>         pagination = payload.get("meta", {}).get("pagination", {})
>         current_page = pagination.get("current_page")
>         total_pages = pagination.get("total_pages")
> 
>         if not current_page or not total_pages or current_page >= total_pages:
>             break
> 
>         page += 1
> 
>     return categories

> 
>     payload = {
>         "name": name,
>         "notes": notes or "",
>     }
> 
>     response = requests.post(url, headers=headers, json=payload, timeout=30)

>     if response.status_code == 302:
>         raise RuntimeError(
>             "Got HTTP 302 redirect. Firefly is sending the API request to login. "
>             "Check FIREFLY_URL, reverse proxy config, and token."

>

>

>

>

>     }
> 
>     existing = set()
>     page = 1
> 
>     while True:
>         response = requests.get(
>             f"{base_url}/api/v1/categories",
>             headers=headers,
>             params={"page": page},
>             timeout=30,
>         )
>         response.raise_for_status()
>         payload = response.json()
> 
>         for item in payload.get("data", []):
>             existing.add(item["attributes"]["name"])
> 
>         pagination = payload.get("meta", {}).get("pagination", {})
>         if pagination.get("current_page") >= pagination.get("total_pages"):
>             break
> 
>         page += 1
> 
>     with open(args.csv_path, newline="", encoding="utf-8-sig") as f:
>         reader = csv.DictReader(f)
> 
>         for row in reader:

>             notes = row.get("notes", "").strip()
> 
>             if not name:
>                 continue
> 
>             if name in existing:
>                 print(f"SKIP existing: {name}")
>                 continue
> 
>             if args.dry_run:

>

> ## Better account seeder script
> 
> Save as:
> 
> `firefly_create_accounts.py`
> 
> ```python
> #!/usr/bin/env python3
> """
> firefly_create_accounts.py
> 
> Creates Firefly III accounts from a CSV file.
> 
> Required CSV columns:
> - name
> - type
> 
> Optional CSV columns:
> - notes
> - account_number
> - iban
> - bic
> - virtual_balance
> - opening_balance
> - opening_balance_date
> - currency_code
> 
> Usage:
>   python3 firefly_create_accounts.py firefly_accounts.csv --dry-run
>   python3 firefly_create_accounts.py firefly_accounts.csv
> """
> 
> import argparse
> import csv
> import os
> import sys
> 
> import requests
> 
> 
> VALID_TYPES = {
>     "asset",
>     "expense",
>     "revenue",
>     "liability",
> }
> 
> 
> def clean(value):
>     if value is None:
>         return ""
>     return str(value).strip()
> 
> 
> def main():
>     parser = argparse.ArgumentParser()
>     parser.add_argument("csv_path")
>     parser.add_argument("--dry-run", action="store_true")
>     args = parser.parse_args()
> 
>     base_url = os.environ.get("FIREFLY_URL", "").rstrip("/")
>     token = os.environ.get("FIREFLY_TOKEN", "")
> 
>     if not base_url:
>         print("ERROR: FIREFLY_URL is not set.", file=sys.stderr)
>         return 1
> 
>     if not token:
>         print("ERROR: FIREFLY_TOKEN is not set.", file=sys.stderr)
>         return 1
> 
>     headers = {
>         "Authorization": f"Bearer {token}",
>         "Accept": "application/json",
>         "Content-Type": "application/json",
>     }
> 
>     existing = set()
>     page = 1
> 
>     while True:
>         response = requests.get(
>             f"{base_url}/api/v1/accounts",
>             headers=headers,
>             params={"page": page},
>             timeout=30,
>         )
>         response.raise_for_status()
>         payload = response.json()
> 
>         for item in payload.get("data", []):
>             existing.add(item["attributes"]["name"])
> 
>         pagination = payload.get("meta", {}).get("pagination", {})
>         if pagination.get("current_page") >= pagination.get("total_pages"):
>             break
> 
>         page += 1
> 
>     with open(args.csv_path, newline="", encoding="utf-8-sig") as f:
>         reader = csv.DictReader(f)
> 
>         for row in reader:
>             name = clean(row.get("name"))
>             account_type = clean(row.get("type")).lower()
> 
>             if not name:
>                 continue
> 
>             if account_type not in VALID_TYPES:
>                 print(f"SKIP invalid type for {name}: {account_type}")
>                 continue
> 
>             if name in existing:
>                 print(f"SKIP existing: {name}")
>                 continue
> 
>             payload = {
>                 "name": name,
>                 "type": account_type,
>             }
> 
>             optional_fields = [
>                 "notes",
>                 "account_number",
>                 "iban",
>                 "bic",
>                 "virtual_balance",
>                 "opening_balance",
>                 "opening_balance_date",
>                 "currency_code",
>             ]
> 
>             for field in optional_fields:
>                 value = clean(row.get(field))
>                 if value:
>                     payload[field] = value
> 
>             if args.dry_run:
>                 print(f"WOULD CREATE: {account_type} :: {name}")
>                 continue
> 
>             response = requests.post(
>                 f"{base_url}/api/v1/accounts",
>                 headers=headers,
>                 json=payload,
>                 timeout=30,
>             )
> 
>             if response.status_code >= 400:
>                 print(f"FAILED: {account_type} :: {name}")
>                 print(response.status_code)
>                 print(response.text)
>                 continue
> 
>             print(f"CREATED: {account_type} :: {name}")
> 
>     return 0
> 
> 
> if __name__ == "__main__":
>     raise SystemExit(main())
> ```
