> # The simplest collection logic
> 
> In `eleventy.config.js`, create collections like:
> 
> ```js
> export default function(eleventyConfig) {
>   eleventyConfig.addCollection("bookChapters", function(collectionApi) {
>     return collectionApi
>       .getFilteredByTag("bookChapter")
>       .sort((a, b) => {
>         return (a.data.order || 0) - (b.data.order || 0);
>       });
>   });
> 
>   eleventyConfig.addCollection("series713", function(collectionApi) {
>     return collectionApi
>       .getAll()
>       .filter(item => item.data.series === "713")
>       .sort((a, b) => {
>         return (a.data.seriesOrder || 999) - (b.data.seriesOrder || 999);
>       });
>   });
> }
> ```
