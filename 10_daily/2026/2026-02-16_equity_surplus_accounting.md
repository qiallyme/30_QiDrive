>         "__pycache__",

>         "node_modules",

>         "dist",

>         "build",

>         ".obsidian",

>         ".venv",

>     def is_dot_dir(path: Path) -> bool:

>     def should_skip_dir(path: Path) -> bool:

>     def iter_children_sorted(dir_path: Path) -> Iterable[Path]:

>         try:

>             kids = list(dir_path.iterdir())

>             return []

>         """
