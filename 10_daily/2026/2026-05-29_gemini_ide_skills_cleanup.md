> 
> ```powershell

> # Scans: C:\Users\codyr\.gemini\config\skills

> 
> $SkillsRoot = "C:\Users\codyr\.gemini\config\skills"

>     Write-Host "Skills folder not found: $SkillsRoot" -ForegroundColor Red

> }
> 
> $skills = Get-ChildItem -Path $SkillsRoot -Directory | ForEach-Object {
>     $skillDir = $_
>     $files = Get-ChildItem -Path $skillDir.FullName -Recurse -File -ErrorAction SilentlyContinue
> 
>     $readme = $files | Where-Object { $_.Name -match '^(README|readme)\.(md|txt)$' } | Select-Object -First 1
>     $manifest = $files | Where-Object { $_.Name -match '(manifest|skill|config|metadata)\.(json|ya?ml|toml|md)$' } | Select-Object -First 1
> 
>     $sampleText = ""
>     if ($readme) {
>         $sampleText = Get-Content $readme.FullName -Raw -ErrorAction SilentlyContinue
>     } elseif ($manifest) {
>         $sampleText = Get-Content $manifest.FullName -Raw -ErrorAction SilentlyContinue
>     }
> 
>     [PSCustomObject]@{
>         SkillName      = $skillDir.Name
>         FullPath       = $skillDir.FullName
>         FileCount      = $files.Count
>         SizeMB         = [Math]::Round((($files | Measure-Object Length -Sum).Sum / 1MB), 2)
>         LastModified   = $skillDir.LastWriteTime
>         HasREADME      = [bool]$readme
>         HasManifest    = [bool]$manifest
>         ManifestFile   = if ($manifest) { $manifest.Name } else { "" }
>         Preview        = ($sampleText -replace '\s+', ' ').Substring(0, [Math]::Min(250, ($sampleText -replace '\s+', ' ').Length))
>     }
> }
> 
> $csvPath = Join-Path $OutDir "gemini_skills_inventory.csv"
> $jsonPath = Join-Path $OutDir "gemini_skills_inventory.json"
> $treePath = Join-Path $OutDir "gemini_skills_tree.txt"
> 
> $skills | Sort-Object SkillName | Export-Csv -Path $csvPath -NoTypeInformation -Encoding UTF8
> $skills | Sort-Object SkillName | ConvertTo-Json -Depth 5 | Out-File -FilePath $jsonPath -Encoding UTF8
> 
> Get-ChildItem -Path $SkillsRoot -Recurse |
>     Select-Object FullName, Length, LastWriteTime |
>     Format-Table -AutoSize |
>     Out-String -Width 300 |
>     Out-File -FilePath $treePath -Encoding UTF8
> 
> Write-Host ""
> Write-Host "Done. Exported Gemini skills audit to:" -ForegroundColor Green
> Write-Host $OutDir -ForegroundColor Cyan
> Write-Host ""
> Write-Host "Files created:"
> Write-Host " - $csvPath"
> Write-Host " - $jsonPath"
